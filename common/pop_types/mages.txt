REPLACE_OR_CREATE:mages = {
	texture = "gfx/interface/icons/pops_icons/mages.dds"
	color = { 0.94 0.51 0.37 }
	start_quality_of_life = 10
	wage_weight = 2
	paid_private_wage = yes
	literacy_target = 0.3
	education_access = 0.3
	dependent_wage = 0.5	# per year
	unemployment = yes
	# 70% - 100% politically engaged
	political_engagement_base = 0.4
	political_engagement_literacy_factor = 0.55
	political_engagement_mult = {
		value = 1
		add = {
			desc = "POP_STARVATION"
			if = {
				limit = {
					is_in_mild_starvation = yes
				}
				value = 0.5
			}
			else_if = {
				limit = {
					is_in_severe_starvation = yes
				}
				value = 1.5
			}
		}
		multiply = {
			desc = "NO_LABOR_MOVEMENT"
			value = 1
			if = {
				limit = {
					owner = {
						NOT = {
							has_technology_researched = labor_movement
						}
					}
				}
				value = 0.75
			}
		}
		multiply = {
			desc = "HAS_POLITICAL_AGITATION"
			value = 1
			if = {
				limit = {
					owner = {
						has_technology_researched = political_agitation
					}
				}
				value = 1.5
			}
		}
	}
	qualifications_growth_desc = "MAGES_QUALIFICATIONS_DESC"
	qualifications = {
		# baseline: ( wealth - 8 ) * (literacy - 0.3) *1.1
		add = {
			value = wealth
			divide = 10
			desc = "QUALIFICATIONS_WEALTH_FACTOR"
		}
		add = {
			value = literacy_rate
			subtract = 0.2
			multiply = 20
			desc = "QUALIFICATIONS_LITERACY_FACTOR"
		}
		if = {
			limit = {
				is_pop_type = laborers
			}
			multiply = {
				value = 1.5
				desc = "QUALIFICATIONS_FAVORED_TYPE"
			}
		}
		if = {
			limit = {
				is_pop_type = machinists
			}
			multiply = {
				value = 2
				desc = "QUALIFICATIONS_FAVORED_TYPE"
			}
		}
	}
	portrait_age = {
		integer_range = {
			min = define:NPortrait|GRACEFUL_AGING_START
			max = define:NPortrait|GRACEFUL_AGING_END
		}
	}
	portrait_pose = {
		value = 0
	}
	portrait_is_female = {
		always = yes
	}
}
